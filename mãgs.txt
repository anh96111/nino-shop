// ===================================================
// HIGHLIGHT SĐT TRÙNG TRONG SHEET
// ===================================================
function highlightDuplicatePhones(sheet) {
  const lastRow = sheet.getLastRow();
  if (lastRow < 2) return;

  const numRows    = lastRow - 1;
  const phoneRange = sheet.getRange(2, 5, numRows, 1);
  const phones     = phoneRange.getValues();

  const count = {};
  phones.forEach(row => {
    const p = row[0].toString().trim();
    if (p) count[p] = (count[p] || 0) + 1;
  });

  for (let i = 0; i < phones.length; i++) {
    const p      = phones[i][0].toString().trim();
    const rowIdx = i + 2;
    if (p && count[p] > 1) {
      sheet.getRange(rowIdx, 1, 1, 20).setBackground("#FFD700");
    } else {
      sheet.getRange(rowIdx, 1, 1, 20).setBackground(null);
    }
  }
}

// ===================================================
// FACEBOOK CAPI
// ===================================================
function callFacebookCAPI(data, orderId, eventName) {
  try {
    const customer = data.customer || {};

    // Normalize phone → E.164 Vietnam
    let normalizedPhone = (customer.phone || data.phone || "").trim().replace(/\D/g, "");
    if (normalizedPhone.startsWith("0")) {
      normalizedPhone = "84" + normalizedPhone.slice(1);
    } else if (normalizedPhone !== "" && !normalizedPhone.startsWith("84")) {
      normalizedPhone = "84" + normalizedPhone;
    }

    const fullName  = (customer.full_name || data.full_name || "").trim();
    const nameParts = fullName.split(" ").filter(Boolean);
    const lastName  = nameParts[0]                 || "";
    const firstName = nameParts.slice(1).join(" ") || lastName;

    const userData = {};

    // Phone hash
    if (data.hashed_phone) {
      userData.ph = data.hashed_phone;
    } else if (normalizedPhone) {
      userData.ph = sha256Hex(normalizedPhone);
    }

    // Name hash — normalize trước khi băm (fix EMQ)
    if (firstName) userData.fn = sha256Hex(firstName.trim().toLowerCase());
    if (lastName)  userData.ln = sha256Hex(lastName.trim().toLowerCase());

    // Browser signals
    if (data.user_agent) userData.client_user_agent = data.user_agent;
    if (data.client_ip)  userData.client_ip_address = data.client_ip;
    if (data.fbp)        userData.fbp = data.fbp;

    // fbc — ưu tiên cookie, fallback fbclid
    if (data.fbc) {
      userData.fbc = data.fbc;
    } else if (data.fbclid) {
      userData.fbc = "fb.1." + Math.floor(Date.now() / 1000) + "." + data.fbclid;
    }

    // =====================================================
    // BƯỚC 6 — external_id:
    // external_id_hashed = true  → HTML đã băm → gán thẳng
    // external_id_hashed = false → plain text   → sha256Hex
    // =====================================================
    if (data.external_id) {
      if (data.external_id_hashed === true) {
        userData.external_id = data.external_id;
      } else {
        userData.external_id = sha256Hex(data.external_id.trim().toLowerCase());
      }
    }

    const customData = {
      content_name: data.product_name || "",
      content_ids:  [data.product_id  || ""],
      content_type: "product",
      currency:     data.currency     || "VND",
      value:        eventName === "Purchase"
                      ? (data.total || 0)
                      : (data.value || data.price || 0)
    };

    if (eventName === "Purchase" && orderId) {
      customData.order_id = orderId;
    }

    const eventIdMap = {
      "Purchase":         data.purchase_event_id          || orderId,
      "AddToCart":        data.add_to_cart_event_id       || generateEventId(),
      "InitiateCheckout": data.initiate_checkout_event_id || generateEventId(),
      "ViewContent":      data.view_content_event_id      || generateEventId()
    };
    const eventId = eventIdMap[eventName] || generateEventId();

    const eventPayload = {
      data: [{
        event_name:       eventName,
        event_time:       Math.floor(Date.now() / 1000),
        event_id:         eventId,
        event_source_url: data.event_source_url || "",
        action_source:    "website",
        user_data:        userData,
        custom_data:      customData
      }]
    };

    const url = "https://graph.facebook.com/v19.0/" + FB_PIXEL_ID
              + "/events?access_token=" + FB_ACCESS_TOKEN;

    const response = UrlFetchApp.fetch(url, {
      method:             "post",
      contentType:        "application/json",
      payload:            JSON.stringify(eventPayload),
      muteHttpExceptions: true
    });

    Logger.log("[CAPI " + eventName + "] " + response.getContentText());

  } catch (err) {
    Logger.log("[CAPI ERROR " + eventName + "] " + err.toString());
  }
}

// ===================================================
// SHA256 — normalize trước khi băm (fix EMQ)
// ===================================================
function sha256Hex(str) {
  const bytes = Utilities.computeDigest(
    Utilities.DigestAlgorithm.SHA_256,
    str.trim().toLowerCase(),   // ← normalize
    Utilities.Charset.UTF_8
  );
  return bytes.map(b => {
    const hex = (b & 0xff).toString(16);
    return hex.length === 1 ? "0" + hex : hex;
  }).join("");
}

// ===================================================
// ORDER ID — DHyymmddHHMMSS
// ===================================================
function generateOrderId() {
  const now = new Date();
  return "DH"
    + now.getFullYear().toString().slice(-2)
    + String(now.getMonth() + 1).padStart(2, "0")
    + String(now.getDate()).padStart(2,       "0")
    + String(now.getHours()).padStart(2,   "0")
    + String(now.getMinutes()).padStart(2, "0")
    + String(now.getSeconds()).padStart(2, "0");
}

// ===================================================
// EVENT ID
// ===================================================
function generateEventId() {
  return new Date().getTime().toString(36) + Math.random().toString(36).slice(2);
}
